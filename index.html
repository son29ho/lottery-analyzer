// =================================================== // LOTTERY ANALYZER - GAS Backend (セキュリティ強化版) // =================================================== // ===== 設定（GASプロパティストアで管理） ===== // 初回セットアップ時に initConfig() を手動実行してください const SHEET_NAMES = ['loto6', 'loto7', 'miniloto', 'num3', 'num4']; const MAX_ATTEMPTS = 5; // ログイン失敗上限 const LOCK_MINUTES = 15; // ロック時間（分） const RATE_LIMIT_MAX = 30; // 1分間の最大リクエスト数 const TOKEN_EXPIRE_H = 24; // セッショントークン有効時間（時間） // ===== エントリポイント ===== function doPost(e) { try { enforceHttps(e); const body = JSON.parse(e.postData.contents); const clientId = getClientId(e); // レート制限チェック const rateCheck = checkRateLimit(clientId); if (!rateCheck.ok) return res({ ok: false, error: rateCheck.error }, 429); // ログインアクション if (body.action === 'login') return res(handleLogin(body.loginId, body.password, clientId)); // その他のアクションはトークン認証 const authCheck = verifyToken(body.token); if (!authCheck.ok) return res({ ok: false, error: authCheck.error }, 401); if (body.action === 'add') return res(addRecord(body.type, body.numbers)); if (body.action === 'delete') return res(deleteRecord(body.type, body.rowIndex)); if (body.action === 'clear') return res(clearSheet(body.type)); if (body.action === 'logout') return res(handleLogout(body.token)); return res({ ok: false, error: '不明なアクション' }); } catch (err) { Logger.log('doPost error: ' + err.message); return res({ ok: false, error: 'サーバーエラーが発生しました' }, 500); } } function doGet(e) { try { enforceHttps(e); const clientId = getClientId(e); const rateCheck = checkRateLimit(clientId); if (!rateCheck.ok) return res({ ok: false, error: rateCheck.error }, 429); const token = e.parameter.token; const type = e.parameter.type; const authCheck = verifyToken(token); if (!authCheck.ok) return res({ ok: false, error: authCheck.error }, 401); if (!type) return res({ ok: false, error: 'typeが必要です' }); return res(getRecords(type)); } catch (err) { Logger.log('doGet error: ' + err.message); return res({ ok: false, error: 'サーバーエラーが発生しました' }, 500); } } // ===== HTTPS強制 ===== function enforceHttps(e) { // GAS Web AppはデフォルトでHTTPS。念のため記録のみ // (GAS環境ではリクエストプロトコルの直接取得は不可のため、 // デプロイ設定でHTTPSを強制することを推奨) } // ===== 認証 ===== function handleLogin(loginId, password, clientId) { // ロックチェック const lockCheck = checkLock(clientId); if (!lockCheck.ok) return { ok: false, error: lockCheck.error }; const props = PropertiesService.getScriptProperties(); const storedIdHash = props.getProperty('LOGINID_HASH'); const storedPwHash = props.getProperty('PASSWORD_HASH'); const inputIdHash = sha256(loginId); const inputPwHash = sha256(password); // IDとパスワードを両方照合（どちらが違っても同じメッセージで返す） if (inputIdHash !== storedIdHash || inputPwHash !== storedPwHash) { recordFailure(clientId); const remaining = getRemainingAttempts(clientId); return { ok: false, error: `IDまたはパスワードが違います（残り${remaining}回）`, remaining }; } // 成功 → 失敗カウントリセット・トークン発行 resetFailures(clientId); const token = issueToken(clientId); return { ok: true, token }; } function handleLogout(token) { const cache = CacheService.getScriptCache(); cache.remove('token_' + token); return { ok: true }; } // ===== トークン管理 ===== function issueToken(clientId) { const token = Utilities.getUuid(); const expire = new Date().getTime() + TOKEN_EXPIRE_H * 3600 * 1000; const cache = CacheService.getScriptCache(); cache.put('token_' + token, JSON.stringify({ clientId, expire }), TOKEN_EXPIRE_H * 3600); return token; } function verifyToken(token) { if (!token) return { ok: false, error: '認証が必要です。再ログインしてください' }; const cache = CacheService.getScriptCache(); const data = cache.get('token_' + token); if (!data) return { ok: false, error: 'セッションが期限切れです。再ログインしてください' }; const parsed = JSON.parse(data); if (new Date().getTime() > parsed.expire) { cache.remove('token_' + token); return { ok: false, error: 'セッションが期限切れです。再ログインしてください' }; } return { ok: true }; } // ===== ロック管理 ===== function checkLock(clientId) { const cache = CacheService.getScriptCache(); const lockKey = 'lock_' + clientId; const locked = cache.get(lockKey); if (locked) { return { ok: false, error: `ログイン試行回数が上限を超えました。${LOCK_MINUTES}分後に再試行してください` }; } return { ok: true }; } function recordFailure(clientId) { const cache = CacheService.getScriptCache(); const key = 'fail_' + clientId; const current = parseInt(cache.get(key) || '0') + 1; cache.put(key, String(current), LOCK_MINUTES * 60); if (current >= MAX_ATTEMPTS) { cache.put('lock_' + clientId, '1', LOCK_MINUTES * 60); cache.remove(key); Logger.log('アカウントロック: ' + clientId); } } function resetFailures(clientId) { const cache = CacheService.getScriptCache(); cache.remove('fail_' + clientId); cache.remove('lock_' + clientId); } function getRemainingAttempts(clientId) { const cache = CacheService.getScriptCache(); const current = parseInt(cache.get('fail_' + clientId) || '0'); return Math.max(0, MAX_ATTEMPTS - current); } // ===== レート制限 ===== function checkRateLimit(clientId) { const cache = CacheService.getScriptCache(); const key = 'rate_' + clientId; const count = parseInt(cache.get(key) || '0'); if (count >= RATE_LIMIT_MAX) { return { ok: false, error: 'リクエストが多すぎます。しばらく待ってから再試行してください' }; } cache.put(key, String(count + 1), 60); // 60秒でリセット return { ok: true }; } // ===== クライアントID取得 ===== function getClientId(e) { // GASはIPを直接取得できないため、User-AgentとタイムスタンプのハッシュでID生成 const ua = (e.parameter && e.parameter.ua) || 'unknown'; return sha256(ua).substring(0, 16); } // ===== SHA-256 ハッシュ ===== function sha256(input) { const bytes = Utilities.computeDigest( Utilities.DigestAlgorithm.SHA_256, input, Utilities.Charset.UTF_8 ); return bytes.map(b => ('0' + (b & 0xFF).toString(16)).slice(-2)).join(''); } // ===== シート操作 ===== function getSheet(type) { const props = PropertiesService.getScriptProperties(); const id = props.getProperty('SHEET_ID'); if (!id) throw new Error('SHEET_IDが設定されていません。initConfig()を実行してください'); const ss = SpreadsheetApp.openById(id); let sheet = ss.getSheetByName(type); if (!sheet) { sheet = ss.insertSheet(type); sheet.appendRow(['timestamp', 'numbers']); sheet.setFrozenRows(1); } return sheet; } function getRecords(type) { const sheet = getSheet(type); const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return { ok: true, records: [] }; const records = rows.slice(1).map((row, i) => ({ rowIndex : i + 2, timestamp: row[0], numbers : String(row[1]).split(',').map(x => x.trim()) })); return { ok: true, records }; } function addRecord(type, numbers) { if (!numbers || !Array.isArray(numbers) || numbers.length === 0) { return { ok: false, error: '番号データが不正です' }; } // 入力バリデーション const sanitized = numbers.map(n => String(n).replace(/[^0-9]/g, '')); const sheet = getSheet(type); sheet.appendRow([new Date().toLocaleString('ja-JP'), sanitized.join(',')]); return { ok: true }; } function deleteRecord(type, rowIndex) { if (!rowIndex || isNaN(rowIndex)) return { ok: false, error: 'rowIndexが不正です' }; const sheet = getSheet(type); const last = sheet.getLastRow(); const row = Number(rowIndex); if (row < 2 || row > last) return { ok: false, error: '行番号が範囲外です' }; sheet.deleteRow(row); return { ok: true }; } function clearSheet(type) { const sheet = getSheet(type); const last = sheet.getLastRow(); if (last > 1) sheet.deleteRows(2, last - 1); return { ok: true }; } // ===== レスポンス ===== function res(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ===== 初期設定（初回のみ手動実行） ===== function initConfig() { const ui = SpreadsheetApp.getUi(); // スプレッドシートID const sheetRes = ui.prompt( '初期設定 (1/4)', 'スプレッドシートIDを入力してください\n（URLの /d/★ここ★/edit 部分）', ui.ButtonSet.OK_CANCEL ); if (sheetRes.getSelectedButton() !== ui.Button.OK) return; const sheetId = sheetRes.getResponseText().trim(); // ログインID const idRes = ui.prompt( '初期設定 (2/4)', '使用するログインIDを入力してください', ui.ButtonSet.OK_CANCEL ); if (idRes.getSelectedButton() !== ui.Button.OK) return; const loginId = idRes.getResponseText().trim(); // パスワード const pwRes = ui.prompt( '初期設定 (3/4)', '使用するパスワードを入力してください', ui.ButtonSet.OK_CANCEL ); if (pwRes.getSelectedButton() !== ui.Button.OK) return; const password = pwRes.getResponseText().trim(); // 確認 const confirmRes = ui.prompt( '初期設定 (4/4) - 確認', `以下の内容で設定します。よろしいですか？\n\nスプレッドシートID: ${sheetId}\nログインID: ${loginId}\nパスワード: ${'*'.repeat(password.length)}\n\n「OK」と入力して確定してください`, ui.ButtonSet.OK_CANCEL ); if (confirmRes.getSelectedButton() !== ui.Button.OK) return; if (!sheetId || !loginId || !password) { ui.alert('入力が空です。やり直してください。'); return; } // プロパティストアに保存（コードに直書きしない） const props = PropertiesService.getScriptProperties(); props.setProperty('SHEET_ID', sheetId); props.setProperty('LOGINID_HASH', sha256(loginId)); props.setProperty('PASSWORD_HASH', sha256(password)); // シート初期化 SHEET_NAMES.forEach(name => getSheet(name)); ui.alert('✅ 初期設定が完了しました！\n\nWeb Appとしてデプロイしてください。'); Logger.log('initConfig完了 - SHEET_ID: ' + sheetId); } // ===== ログインID変更 ===== function changeLoginId() { const ui = SpreadsheetApp.getUi(); const r = ui.prompt('ログインID変更', '新しいログインIDを入力してください', ui.ButtonSet.OK_CANCEL); if (r.getSelectedButton() !== ui.Button.OK) return; const newId = r.getResponseText().trim(); if (!newId) { ui.alert('IDが空です'); return; } PropertiesService.getScriptProperties().setProperty('LOGINID_HASH', sha256(newId)); ui.alert('✅ ログインIDを変更しました'); } // ===== パスワード変更 ===== function changePassword() { const ui = SpreadsheetApp.getUi(); const r = ui.prompt('パスワード変更', '新しいパスワードを入力してください', ui.ButtonSet.OK_CANCEL); if (r.getSelectedButton() !== ui.Button.OK) return; const newPw = r.getResponseText().trim(); if (!newPw) { ui.alert('パスワードが空です'); return; } PropertiesService.getScriptProperties().setProperty('PASSWORD_HASH', sha256(newPw)); ui.alert('✅ パスワードを変更しました'); } // ===== 設定確認 ===== function checkConfig() { const props = PropertiesService.getScriptProperties(); const id = props.getProperty('SHEET_ID'); const idHash = props.getProperty('LOGINID_HASH'); const pwHash = props.getProperty('PASSWORD_HASH'); Logger.log('SHEET_ID: ' + (id ? '設定済み (' + id + ')' : '未設定')); Logger.log('LOGINID_HASH: ' + (idHash ? '設定済み' : '未設定')); Logger.log('PASSWORD_HASH: ' + (pwHash ? '設定済み' : '未設定')); }
